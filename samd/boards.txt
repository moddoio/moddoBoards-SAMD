# "USB (CDC & HID)" (default) uses the BOOTLOADER's USB driver via the
# bl_cdc_export table - no USB stack is compiled into the application.
# Serial (CDC) always works; HID composite support links in automatically
# when the sketch uses the HID library (device then enumerates with PID
# 0xE44E instead of the bootloader's 0xE44B). It REQUIRES the matching
# bootloader (shipped in bootloaders/pinch/, flashed via Burn Bootloader);
# on an older bootloader the USB port stays inert by design.
# "USB disabled" removes USB entirely (low-power builds; Serial = Serial1).
pinch.menu.usb.cdc=USB (CDC & HID)
pinch.menu.usb.cdc.build.usbcom=CDC_BOOTLOADER
pinch.menu.usb.cdc.build.pid=0xE44B
pinch.menu.usb.cdc.build.ldscript=flash_16KB_blcdc.ld
pinch.menu.usb.none=USB disabled
pinch.menu.usb.none.build.usbcom=USB_DISABLED
pinch.menu.usb.none.build.pid=0xE44C
